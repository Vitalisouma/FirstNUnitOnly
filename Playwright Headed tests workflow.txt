  name: Playwright Headed Tests on Windows

    on:
      push:
        branches: [ main, master ]
      pull_request:
        branches: [ main, master ]

    jobs:
      test:
        timeout-minutes: 60
        runs-on: windows-latest # Specify Windows environment

        steps:
          - uses: actions/checkout@v4 # Checkout your repository code
          - uses: actions/setup-node@v4 # Setup Node.js
            with:
              node-version: lts/* # Use a Long Term Support version of Node.js

          - name: Install dependencies
            run: npm ci # Install project dependencies

          - name: Install Playwright Browsers
            run: npx playwright install --with-deps # Install Playwright browsers and their dependencies

          - name: Run Playwright headed tests
            run: npx playwright test --headed # Run tests in headed mode

          - uses: actions/upload-artifact@v4 # Upload the Playwright report
            if: always() # Upload even if tests fail
            with:
              name: playwright-report
              path: playwright-report/
              retention-days: 30
